 600    public string CalculateSHA1(string input)
 601    {
 602        // Convert input string to bytes
 603        byte[] data = Encoding.UTF8.GetBytes(input);
 604
 605        // Initialize variables
 606        uint h0 = 0x67452301;
 607        uint h1 = 0xEFCDAB89;
 608        uint h2 = 0x98BADCFE;
 609        uint h3 = 0x10325476;
 610        uint h4 = 0xC3D2E1F0;
 611
 612        // Pre-processing: append padding bits and length
 613        long originalLength = data.Length * 8;
 614        int paddingLength = 64 - ((data.Length + 8) % 64); // Adjust padding calculation
 615        if (paddingLength == 0)
 616            paddingLength = 64;
 617
 618        byte[] padding = new byte[paddingLength];
 619        padding[0] = 0x80;
 620
 621        // Append a single '1' bit
 622        Array.Resize(ref data, data.Length + paddingLength);
 623        Array.Copy(padding, 0, data, data.Length - paddingLength, paddingLength);
 624
 625        // Append original length in bits as a 64-bit big-endian integer
 626        byte[] lengthBytes = BitConverter.GetBytes(originalLength);
 627        Array.Resize(ref data, data.Length + 8);
 628        Array.Copy(lengthBytes, 0, data, data.Length - 8, 8);
 629
 630        // Process the message in successive 512-bit chunks
 631        for (int i = 0; i < data.Length; i += 64)
 632        {
 633            uint[] w = new uint[80];
 634
 635            // Break chunk into sixteen 32-bit big-endian words w[i]
 636            for (int j = 0; j < 16; j++)
 637            {
 638                w[j] = (uint)((data[i + j * 4] << 24) |
 639                               (data[i + j * 4 + 1] << 16) |
 640                               (data[i + j * 4 + 2] << 8) |
 641                               (data[i + j * 4 + 3]));
 642            }
 643
 644            // Extend the sixteen 32-bit words into eighty 32-bit words
 645            for (int j = 16; j < 80; j++)
 646            {
 647                w[j] = (w[j - 3] ^ w[j - 8] ^ w[j - 14] ^ w[j - 16]);
 648                w[j] = (w[j] << 1) | (w[j] >> 31); // Left rotate by 1 bit
 649            }
 650
 651            // Initialize hash value for this chunk
 652            uint a = h0;
 653            uint b = h1;
 654            uint c = h2;
 655            uint d = h3;
 656            uint e = h4;
 657
 658            // Main loop
 659            for (int j = 0; j < 80; j++)
 660            {
 661                uint f, k;
 662                if (j < 20)
 663                {
 664                    f = (b & c) | ((~b) & d);
 665                    k = 0x5A827999;
 666                }
 667                else if (j < 40)
 668                {
 669                    f = b ^ c ^ d;
 670                    k = 0x6ED9EBA1;
 671                }
 672                else if (j < 60)
 673                {
 674                    f = (b & c) | (b & d) | (c & d);
 675                    k = 0x8F1BBCDC;
 676                }
 677                else
 678                {
 679                    f = b ^ c ^ d;
 680                    k = 0xCA62C1D6;
 681                }
 682
 683                uint temp = ((a << 5) | (a >> 27)) + f + e + k + w[j];
 684                e = d;
 685                d = c;
 686                c = ((b << 30) | (b >> 2));
 687                b = a;
 688                a = temp;
 689            }
 690
 691            // Add this chunk's hash to result so far
 692            h0 += a;
 693            h1 += b;
 694            h2 += c;
 695            h3 += d;
 696            h4 += e;
 697        }
 698
 699        // Produce the final hash value (big-endian)
 700        byte[] hashBytes = new byte[20];
 701        Array.Copy(BitConverter.GetBytes(h0), 0, hashBytes, 0, 4);
 702        Array.Copy(BitConverter.GetBytes(h1), 0, hashBytes, 4, 4);
 703        Array.Copy(BitConverter.GetBytes(h2), 0, hashBytes, 8, 4);
 704        Array.Copy(BitConverter.GetBytes(h3), 0, hashBytes, 12, 4);
 705        Array.Copy(BitConverter.GetBytes(h4), 0, hashBytes, 16, 4);
 706
 707        // Convert hash bytes to string
 708        string hashString = BitConverter.ToString(hashBytes).Replace("-", "").ToLower();
 709
 710        return hashString;
 711    }

 763/// <summary>
 764/// Finds the longest common subsequence (LCS) of two input strings.
 765/// </summary>
 766/// <param name="text1">The first input string.</param>
 767/// <param name="text2">The second input string.</param>
 768/// <returns>The longest common subsequence of the input strings.</returns>
 769public static string LCS(string text1, string text2)
 770{
 771    int m = text1.Length;
 772    int n = text2.Length;
 773    // Create a 2D array to store the length of the LCS for substrings of text1 and text2
 774    int[,] dp = new int[m + 1, n + 1];
 775
 776    // Fill the dp table using bottom-up dynamic programming approach
 777    for (int i = 1; i <= m; ++i)
 778    {
 779        for (int j = 1; j <= n; ++j)
 780        {
 781            if (text1[i - 1] == text2[j - 1])
 782            {
 783                dp[i, j] = dp[i - 1, j - 1] + 1;
 784            }
 785            else
 786            {
 787                dp[i, j] = Math.Max(dp[i - 1, j], dp[i, j - 1]);
 788            }
 789        }
 790    }
 791
 792    // Reconstruct the LCS from dp table
 793    int x = m, y = n;
 794    string lcs = "";
 795
 796    while (x > 0 && y > 0)
 797    {
 798        if (text1[x - 1] == text2[y - 1])
 799        {
 800            lcs = text1[x - 1] + lcs;
 801            --x;
 802            --y;
 803        }
 804        else if (dp[x - 1, y] > dp[x, y - 1])
 805        {
 806            --x;
 807        }
 808        else
 809        {
 810            --y;
 811        }
 812    }
 813
 814    return lcs;
 815}
